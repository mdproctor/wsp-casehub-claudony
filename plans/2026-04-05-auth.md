# Authentication Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add passkey (WebAuthn) authentication for browser users and API key authentication for the Agent, with invite-link-based registration.

**Architecture:** `quarkus-security-webauthn` handles the WebAuthn ceremony and session cookie. A custom `HttpAuthenticationMechanism` accepts `X-Api-Key` as an alternative credential for machine clients. `InviteService` gates registration to invite-link holders only (skipped for the very first user when the credential store is empty).

**Tech Stack:** Quarkus 3.9.5, `quarkus-security-webauthn`, `quarkus-test-security`, Jackson for JSON file storage, WebAuthn browser API (no external JS libraries)

```java
package dev.remotecc.server.auth;

import jakarta.enterprise.context.ApplicationScoped;
import java.time.Instant;
import java.time.Duration;
import java.util.UUID;
import java.util.concurrent.ConcurrentHashMap;

@ApplicationScoped
public class InviteService {

    static final Duration TTL = Duration.ofHours(24);

    private final ConcurrentHashMap<String, Instant> tokens = new ConcurrentHashMap<>();

    public String generate() {
        var token = UUID.randomUUID().toString();
        tokens.put(token, Instant.now().plus(TTL));
        return token;
    }

    public boolean isValid(String token) {
        var expiry = tokens.get(token);
        return expiry != null && Instant.now().isBefore(expiry);
    }

    public void consume(String token) {
        tokens.remove(token);
    }

    /** Test hook — inject a token with a custom expiry. */
    void injectForTest(String token, Instant expiry) {
        tokens.put(token, expiry);
    }
}
```

```java
package dev.remotecc.server.auth;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.type.CollectionType;
import dev.remotecc.config.RemoteCCConfig;
import io.quarkus.security.webauthn.WebAuthnCredentialRecord;
import io.quarkus.security.webauthn.WebAuthnUserProvider;
import io.smallrye.mutiny.Uni;
import io.smallrye.mutiny.infrastructure.Infrastructure;
import jakarta.enterprise.context.ApplicationScoped;
import jakarta.inject.Inject;
import java.io.IOException;
import java.nio.file.*;
import java.util.ArrayList;
import java.util.List;

@ApplicationScoped
public class CredentialStore implements WebAuthnUserProvider {

    record StoredCredential(
        String username,
        String credentialId,
        String publicKey,
        int publicKeyAlgorithm,
        long counter,
        String aaguid
    ) {}

    private static final ObjectMapper MAPPER = new ObjectMapper();
    private static final CollectionType LIST_TYPE =
        MAPPER.getTypeFactory().constructCollectionType(List.class, StoredCredential.class);

    private final RemoteCCConfig config;

    @Inject
    public CredentialStore(RemoteCCConfig config) {
        this.config = config;
    }

    public boolean isEmpty() {
        return readAll().isEmpty();
    }

    @Override
    public Uni<List<WebAuthnCredentialRecord>> findByUsername(String username) {
        return Uni.createFrom()
            .item(() -> readAll().stream()
                .filter(c -> c.username().equals(username))
                .map(this::toRecord)
                .toList())
            .runSubscriptionOn(Infrastructure.getDefaultWorkerPool());
    }

    @Override
    public Uni<List<WebAuthnCredentialRecord>> findByCredentialId(String credentialId) {
        return Uni.createFrom()
            .item(() -> readAll().stream()
                .filter(c -> c.credentialId().equals(credentialId))
                .map(this::toRecord)
                .toList())
            .runSubscriptionOn(Infrastructure.getDefaultWorkerPool());
    }

    @Override
    public Uni<Void> store(WebAuthnCredentialRecord record) {
        return Uni.createFrom()
            .item(() -> {
                var all = readAll();
                // Remove any existing entry with same credentialId (re-registration)
                all.removeIf(c -> c.credentialId().equals(record.getCredentialID()));
                all.add(new StoredCredential(
                    record.getUserName(),
                    record.getCredentialID(),
                    record.getPublicKey(),
                    (int) record.getPublicKeyAlgorithm(),
                    record.getCounter(),
                    record.getAaguid()
                ));
                writeAll(all);
                return (Void) null;
            })
            .runSubscriptionOn(Infrastructure.getDefaultWorkerPool());
    }

    @Override
    public Uni<Void> update(String credentialId, long counter) {
        return Uni.createFrom()
            .item(() -> {
                var all = readAll();
                for (int i = 0; i < all.size(); i++) {
                    var c = all.get(i);
                    if (c.credentialId().equals(credentialId)) {
                        all.set(i, new StoredCredential(
                            c.username(), c.credentialId(), c.publicKey(),
                            c.publicKeyAlgorithm(), counter, c.aaguid()
                        ));
                        break;
                    }
                }
                writeAll(all);
                return (Void) null;
            })
            .runSubscriptionOn(Infrastructure.getDefaultWorkerPool());
    }

    // ---- private file I/O ----

    private synchronized List<StoredCredential> readAll() {
        var path = credentialsPath();
        if (!Files.exists(path)) return new ArrayList<>();
        try {
            return new ArrayList<>(MAPPER.readValue(path.toFile(), LIST_TYPE));
        } catch (IOException e) {
            return new ArrayList<>();
        }
    }

    private synchronized void writeAll(List<StoredCredential> creds) {
        var path = credentialsPath();
        var tmp = Path.of(path + ".tmp");
        try {
            Files.createDirectories(path.getParent());
            MAPPER.writeValue(tmp.toFile(), creds);
            Files.move(tmp, path, StandardCopyOption.REPLACE_EXISTING, StandardCopyOption.ATOMIC_MOVE);
        } catch (IOException e) {
            throw new RuntimeException("Failed to write credentials file: " + path, e);
        }
    }

    private Path credentialsPath() {
        return Path.of(config.credentialsFile().replace("${user.home}", System.getProperty("user.home")));
    }

    private WebAuthnCredentialRecord toRecord(StoredCredential c) {
        var r = new WebAuthnCredentialRecord();
        r.setUserName(c.username());
        r.setCredentialID(c.credentialId());
        r.setPublicKey(c.publicKey());
        r.setPublicKeyAlgorithm(c.publicKeyAlgorithm());
        r.setCounter(c.counter());
        r.setAaguid(c.aaguid());
        return r;
    }

    /** Test hook — write a raw credential entry directly to the JSON file. */
    void writeForTest(String username, String credId, String pubKey, int alg, long counter, String aaguid) {
        var all = readAll();
        all.add(new StoredCredential(username, credId, pubKey, alg, counter, aaguid));
        writeAll(all);
    }
}
```

**If you get compile errors on `WebAuthnCredentialRecord` setters** (e.g., `setPublicKey` doesn't exist), open the class in your IDE and adjust the setter names in `toRecord()` and the getter names in `store()` to match. The fields to persist are: username, credentialId, publicKey (base64url), algorithm (COSE int), counter, aaguid.

```java
package dev.remotecc.server.auth;

import dev.remotecc.config.RemoteCCConfig;
import io.netty.handler.codec.http.HttpResponseStatus;
import io.quarkus.security.identity.IdentityProviderManager;
import io.quarkus.security.identity.SecurityIdentity;
import io.quarkus.security.identity.request.AuthenticationRequest;
import io.quarkus.security.runtime.QuarkusPrincipal;
import io.quarkus.security.runtime.QuarkusSecurityIdentity;
import io.quarkus.vertx.http.runtime.security.ChallengeData;
import io.quarkus.vertx.http.runtime.security.HttpAuthenticationMechanism;
import io.quarkus.vertx.http.runtime.security.HttpCredentialTransport;
import io.smallrye.mutiny.Uni;
import io.vertx.ext.web.RoutingContext;
import jakarta.enterprise.context.ApplicationScoped;
import jakarta.inject.Inject;
import java.util.Optional;
import java.util.Set;

@ApplicationScoped
public class ApiKeyAuthMechanism implements HttpAuthenticationMechanism {

    @Inject
    RemoteCCConfig config;

    @Override
    public Uni<SecurityIdentity> authenticate(RoutingContext context, IdentityProviderManager identityProviderManager) {
        var apiKey = context.request().getHeader("X-Api-Key");
        if (apiKey == null) {
            // No API key present — let other mechanisms (WebAuthn cookie) try
            return Uni.createFrom().optional(Optional.empty());
        }
        var expected = config.agentApiKey();
        if (expected.isEmpty() || !apiKey.equals(expected.get())) {
            return Uni.createFrom().failure(
                new io.quarkus.security.AuthenticationFailedException("Invalid API key"));
        }
        return Uni.createFrom().item(
            QuarkusSecurityIdentity.builder()
                .setPrincipal(new QuarkusPrincipal("agent"))
                .addRole("user")
                .build());
    }

    @Override
    public Uni<ChallengeData> getChallenge(RoutingContext context) {
        return Uni.createFrom().item(
            new ChallengeData(HttpResponseStatus.UNAUTHORIZED.code(), null, null));
    }

    @Override
    public Set<Class<? extends AuthenticationRequest>> getCredentialTypes() {
        return Set.of();
    }

    @Override
    public Uni<HttpCredentialTransport> getCredentialTransport(RoutingContext context) {
        return Uni.createFrom().optional(Optional.empty());
    }
}
```

```java
package dev.remotecc.server.auth;

import io.quarkus.security.Authenticated;
import jakarta.inject.Inject;
import jakarta.ws.rs.*;
import jakarta.ws.rs.core.*;
import org.jboss.logging.Logger;

@Path("/auth")
public class AuthResource {

    private static final Logger LOG = Logger.getLogger(AuthResource.class);

    @Inject InviteService inviteService;
    @Inject CredentialStore credentialStore;

    @POST
    @Path("/invite")
    @Authenticated
    @Produces(MediaType.APPLICATION_JSON)
    public Response generateInvite(@Context UriInfo uriInfo) {
        var token = inviteService.generate();
        var url = uriInfo.getBaseUriBuilder()
            .path("auth/register")
            .queryParam("token", token)
            .build()
            .toString();
        LOG.infof("Generated invite link (token=%s...)", token.substring(0, 8));
        return Response.ok("{\"url\":\"" + url + "\"}").build();
    }

    @GET
    @Path("/register")
    @Produces(MediaType.TEXT_HTML)
    public Response registerPage(@QueryParam("token") String token) {
        // First user: no credentials registered yet — allow registration without invite
        if (credentialStore.isEmpty()) {
            return serveRegisterPage();
        }
        // Subsequent users: require a valid invite token
        if (token == null || !inviteService.isValid(token)) {
            LOG.warnf("Invalid or missing invite token: %s", token);
            return Response.status(403)
                .entity("<html><body><h2>This invite has expired — ask for a new one.</h2></body></html>")
                .type(MediaType.TEXT_HTML)
                .build();
        }
        return serveRegisterPage();
    }

    private Response serveRegisterPage() {
        // The register.html is served as a static resource but we need to control access.
        // Forward to the static file by redirecting internally.
        // The actual HTML is at /auth/register.html — we read and return it here
        // so that the /auth/register path (with token param) is what gets visited.
        try {
            var stream = getClass().getResourceAsStream("/META-INF/resources/auth/register.html");
            if (stream == null) {
                return Response.serverError().entity("register.html not found").build();
            }
            var html = new String(stream.readAllBytes());
            return Response.ok(html).type(MediaType.TEXT_HTML).build();
        } catch (Exception e) {
            return Response.serverError().build();
        }
    }

    @GET
    @Path("/login")
    @Produces(MediaType.TEXT_HTML)
    public Response loginPage() {
        try {
            var stream = getClass().getResourceAsStream("/META-INF/resources/auth/login.html");
            if (stream == null) return Response.serverError().entity("login.html not found").build();
            return Response.ok(new String(stream.readAllBytes())).type(MediaType.TEXT_HTML).build();
        } catch (Exception e) {
            return Response.serverError().build();
        }
    }
}
```

```java
package dev.remotecc.agent;

import dev.remotecc.config.RemoteCCConfig;
import jakarta.inject.Inject;
import jakarta.ws.rs.client.ClientRequestContext;
import jakarta.ws.rs.client.ClientRequestFilter;
import jakarta.ws.rs.ext.Provider;
import java.io.IOException;

@Provider
public class ApiKeyClientFilter implements ClientRequestFilter {

    @Inject
    RemoteCCConfig config;

    @Override
    public void filter(ClientRequestContext requestContext) throws IOException {
```
